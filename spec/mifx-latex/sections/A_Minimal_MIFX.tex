\appendix
\section{Minimal MIFX Package}

This appendix shows the smallest possible valid MIFX package.

\subsection{Directory Structure}

\begin{verbatim}
job.mifx
|
+-- job.json
|
+-- setups/
|   \-- set-1/
|       +-- setup.json
|       \-- operations/
|           +-- op-1.json
|           \-- cldata/
|               \-- op-1.apt
|
\-- tools/
    +-- main.json
    \-- main/
        \-- 1.json
\end{verbatim}

\subsection{job.json}

\begin{verbatim}
{
  "mifxVersion": "0.1",
  "header": {
    "job": {
      "name": "Minimal Example",
      "units": {
        "system": "metric",
        "linear": "mm"
      }
    }
  },
  "setups": [
    {
      "id": "set-1",
      "path": "setups/set-1/setup.json"
    }
  ],
  "tools": [
    {
      "role": "tool_group",
      "kind": "json",
      "path": "tools/main.json",
      "present": true
    }
  ]
}
\end{verbatim}

\subsection{setup.json}

\begin{verbatim}
{
  "id": "set-1",
  "index": 1,
  "name": "Example Setup",
  "operations": [
    {
      "id": "op-1",
      "role": "toolpath",
      "kind": "mill",
      "path": "operations/op-1.json"
    }
  ]
}
\end{verbatim}

\subsection{op-1.json}

\begin{verbatim}
{
  "id": "op-1",
  "name": "Face",
  "strategy": "face",
  "artifacts": [
    {
      "role": "tool",
      "kind": "json",
      "path": "../../../tools/main/1.json",
      "present": true
    },
    {
      "role": "toolpath",
      "kind": "apt",
      "path": "cldata/op-1.apt",
      "present": true
    }
  ]
}
\end{verbatim}
