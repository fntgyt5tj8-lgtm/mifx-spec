\appendix
\section{Minimal MIFX Package}

This appendix shows the smallest possible valid MIFX package.

\subsection{Directory Structure}

\begin{verbatim}
job.mifx
|
+-- job.json
|
+-- setups/
|   \-- set-1/
|       +-- setup.json
|       \-- operations/
|           +-- op-1.json
|           \-- cldata/
|               \-- op-1.apt
|
\-- tools/
    \-- main/
        \-- 1.json
\end{verbatim}

\subsection{job.json}

\begin{verbatim}
{
  "mifxVersion": "0.1",
  "header": {
    "job": {
      "name": "Minimal Example",
      "units": {
        "system": "metric",
        "linear": "mm"
      }
    }
  }
}
\end{verbatim}

\subsection{setup.json}

\begin{verbatim}
{
  "id": "set-1",
  "index": 1,
  "name": "Example Setup"
}
\end{verbatim}

\subsection{op-1.json}

\begin{verbatim}
{
  "id": "op-1",
  "kind": "mill",
  "name": "Face",
  "toolRef": "1",
  "artifacts": [
    {
      "role": "toolpath",
      "kind": "apt",
      "path": "cldata/op-1.apt",
      "present": true
    }
  ]
}
\end{verbatim}
